```plantuml

node "Компьютер пользователя" <<device>> {
	node "Браузер" {
		[Расширение для браузера]
		[Веб-приложение]
	}
}

node "Сервер приложения" <<device>> {
	node "Docker" {
		node "Container1" {
			[API приложения]
			database "База данных аутентификации" {
				[PostgreSQL]
			}
		}
	
		node "Container2" {
			[API работы с узлами]
			database "База данных иерархической структуры узлов" {
				[Neo4j]
			}
		}
	}
}

[Веб-приложение] --> [Расширение для браузера] : Встраивается
[Расширение для браузера] --> [API приложения] : HTTP
[API приложения] -> [API работы с узлами] : Перенаправляет запросы
[API приложения] --> [PostgreSQL] : Читает и записывает данные
[API работы с узлами] --> [Neo4j] : Читает и записывает данные
```
